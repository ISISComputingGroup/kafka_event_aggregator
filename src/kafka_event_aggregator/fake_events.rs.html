3use flatbuffers::FlatBufferBuilder;
4use isis_streaming_data_types::flatbuffers_generated::events_ev44::{

9};
10use rand::RngExt;
11use rand::rngs::ChaCha8Rng;
12use std::iter;

14/// Make fake flatbuffers-encoded events
15///
16/// This is only used in tests and benchmarks.
17pub fn make_fake_flatbuffers_encoded_events(
18    rng: &mut ChaCha8Rng,
19    reference_time: i64,
20    num_messages: usize,
21    events_per_message: usize,
22) -> Vec<Vec<u8>> {
23    (0..num_messages)
24        .map(|_| {
25            let mut pixel_ids = Vec::with_capacity(events_per_message);
26            let mut tofs = Vec::with_capacity(events_per_message);
27
28            for _ in 0..events_per_message {
29                pixel_ids.push(rng.random_range(0..65536));
30                tofs.push(rng.random_range(0..100_000_000));
31            }
32
33            let mut fbb = FlatBufferBuilder::new();
34            let args = Event44MessageArgs {
35                source_name: Some(fbb.create_string("source_name")),
36                message_id: 0,
37                reference_time: Some(fbb.create_vector(&[reference_time])),
38                reference_time_index: Some(fbb.create_vector(&[0])),
39                time_of_flight: Some(fbb.create_vector(&tofs)),
40                pixel_id: Some(fbb.create_vector(&pixel_ids)),
41            };
42
43            let ev44 = Event44Message::create(&mut fbb, &args);
44            finish_event_44_message_buffer(&mut fbb, ev44);
45            fbb.finished_data().to_owned()
46        })
47        .chain(iter::once_with(|| {
48            let mut fbb = FlatBufferBuilder::new();
49            let args = Pu00MessageArgs {
50                source_name: Some(fbb.create_string("source_name")),
51                message_id: 0,
52                reference_time,
53                proton_charge: Some(1.2345),
54                vetos: Some(0b11010011),
55                period_number: Some(1),
56            };
57            let pu00 = Pu00Message::create(&mut fbb, &args);
58            finish_pu_00_message_buffer(&mut fbb, pu00);
59            fbb.finished_data().to_owned()
60        }))
61        .collect()
62}

64#[cfg(test)]
65mod tests {
66    use super::*;
67    use isis_streaming_data_types::{DeserializedMessage, deserialize_message};
68    use rand::SeedableRng;
69    use rand::rngs::ChaCha8Rng;
70
71    #[test]
72    fn test_fake_events_deserialize() {
73        let mut rng = ChaCha8Rng::seed_from_u64(0);
74        let events = make_fake_flatbuffers_encoded_events(&mut rng, 20_000_000, 1, 5);
75
76        assert_eq!(events.len(), 2); // ev44 followed by pu00
77
78        match deserialize_message(events.get(0).unwrap()) {
79            Ok(DeserializedMessage::EventDataEv44(data)) => {
80                assert_eq!(data.reference_time().get(0), 20_000_000);
81                assert_eq!(data.reference_time_index().get(0), 0);
82                assert_eq!(data.time_of_flight().unwrap().len(), 5);
83                assert_eq!(data.pixel_id().unwrap().len(), 5);
84            }
85            _ => panic!("did not deserialize as expected"),
86        }
87
88        match deserialize_message(events.get(1).unwrap()) {
89            Ok(DeserializedMessage::PulseMetadataPu00(data)) => {
90                assert_eq!(data.reference_time(), 20_000_000);
91            }
92            _ => panic!("did not deserialize as expected"),
93        }
94    }
