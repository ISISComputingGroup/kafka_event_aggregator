13pub fn make_consumer(config: &AggregatorConfig) -> Result<StreamConsumer<DefaultConsumerContext>> {
14    let mut client_config = ClientConfig::new();
15
16    for (k, v) in config.kafka_consumer_settings() {
17        client_config.set(k, v);
18    }
19
20    let consumer: StreamConsumer<DefaultConsumerContext> = client_config.create()?;
21
22    consumer
23        .subscribe(&[config.input_topic()])
24        .with_context(|| format!("Failed to subscribe to topic {}", config.input_topic()))?;
25
26    Ok(consumer)
27}

29fn latest_message_on_one_partition(
30    config: &AggregatorConfig,
31    consumer: &BaseConsumer<DefaultConsumerContext>,
32    partition: i32,
33) -> Result<i64> {
34    let (low, high) = consumer.fetch_watermarks(
35        config.output_topic(),
36        partition,
37        Duration::from_millis(config.read_last_message_timeout_ms()),
38    )?;
39
40    if high <= low {
41        bail!(
42            "Partition {} on topic {} has no messages; skipping",
43            partition,
44            config.output_topic()
45        );
46    }
47
48    let mut tpl = TopicPartitionList::new();
49    tpl.add_partition_offset(config.output_topic(), partition, Offset(high - 1))?;
50    consumer.assign(&tpl)?;
51
52    let mut attempts = 0;
53
54    // Note: must call poll repeatedly with short timeouts, rather than calling poll
55    // once with a long timeout here, to avoid race condition in kafka with polling
56    // immediately after assignment.
57    while attempts < config.read_last_message_timeout_ms() {
58        if let Some(msg) = consumer.poll(Duration::from_millis(1)) {
59            return match msg {
60                Ok(message) => {
61                    if let Some(payload) = message.payload() {
62                        match deserialize_message(payload) {
63                            Ok(DeserializedMessage::EventDataEv44(msg)) => Ok(msg.message_id()),
64                            Ok(DeserializedMessage::PulseMetadataPu00(msg)) => Ok(msg.message_id()),
65                            _ => {
66                                bail!(
67                                    "Cannot deserialize latest message on partition {}",
68                                    partition
69                                );
70                            }
71                        }
72                    } else {
73                        bail!("Latest message has no payload on partition {}", partition);
74                    }
75                }
76                Err(err) => {
77                    bail!(
78                        "Cannot read message from kafka: {} on partition {}",
79                        err,
80                        partition
81                    );
82                }
83            };
84        }
85        attempts += 1;
86    }
87
88    bail!("Could not read any messages from partition {}", partition)
89}

91pub fn get_most_recent_message_id(config: &AggregatorConfig) -> Result<i64> {
92    let mut client_config = ClientConfig::new();
93
94    for (k, v) in config.kafka_consumer_settings() {
95        client_config.set(k, v);
96    }
97
98    let consumer: BaseConsumer<DefaultConsumerContext> = client_config.create()?;
99
100    let metadata = consumer.fetch_metadata(Some(config.output_topic()), Duration::from_secs(10))?;
101
102    metadata
103        .topics()
104        .iter()
105        .find(|t| t.name() == config.output_topic())
106        .with_context(|| "Cannot get topic")?
107        .partitions()
108        .iter()
109        .filter_map(
110            |p| match latest_message_on_one_partition(config, &consumer, p.id()) {
111                Err(err) => {
112                    warn!("{}", err);
113                    None
114                }
115                Ok(val) => {
116                    debug!("Latest id for partition {} is {}", p.id(), val);
117                    Some(val)
118                }
119            },
120        )
121        .max()
122        .with_context(|| format!("No usable messages on topic {}", config.output_topic()))
123}

125pub fn make_producer(
126    config: &AggregatorConfig,
127) -> Result<ThreadedProducer<DefaultProducerContext>> {
128    let mut client_config = ClientConfig::new();
129
130    // Check that enable.idempotence has been set - if not, fail loudly as
131    // this is not a valid configuration for `kafka_event_aggregator` and the stream format
132    // will be incorrect.
133    if config.kafka_producer_settings().get("enable.idempotence") != Some(&"true".to_owned()) {
134        bail!("Kafka producer is not set to be idempotent.
135It is not possible to produce a coherent stream with a non-idempotent producer as messages may be duplicated, missing, or out of order (within a single partition).
136See documentation of this program for details of the _events stream format.
137")
138    }
139
140    for (k, v) in config.kafka_producer_settings() {
141        client_config.set(k, v);
142    }
143
144    Ok(client_config.create()?)
145}
